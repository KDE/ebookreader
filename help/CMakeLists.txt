SET(DOCUMENT_NAME tabletReaderHelp)

PROJECT(Document)
FIND_PACKAGE(LATEX)

IF(LATEX_COMPILER)
	ADD_CUSTOM_COMMAND(OUTPUT ${Document_BINARY_DIR}/${DOCUMENT_NAME}.dvi
			   COMMAND ${LATEX_COMPILER}
			   ARGS ${Document_SOURCE_DIR}/${DOCUMENT_NAME}.tex
			   DEPENDS ${Document_SOURCE_DIR}/${DOCUMENT_NAME}.tex
		      	   COMMENT "Tex2dvi")

IF(DVIPS_CONVERTER)
	ADD_CUSTOM_COMMAND(OUTPUT ${Document_BINARY_DIR}/${DOCUMENT_NAME}.ps
			   COMMAND ${DVIPS_CONVERTER}
			   ARGS ${Document_BINARY_DIR}/${DOCUMENT_NAME}.dvi
			   -o ${Document_BINARY_DIR}/${DOCUMENT_NAME}.ps
			   DEPENDS ${Document_BINARY_DIR}/${DOCUMENT_NAME}.dvi
			   COMMENT "dvi2ps")

IF(PS2PDF_CONVERTER)
	ADD_CUSTOM_COMMAND(OUTPUT ${Document_BINARY_DIR}/${DOCUMENT_NAME}.pdf
        		   COMMAND ${PS2PDF_CONVERTER}
			   ARGS ${Document_BINARY_DIR}/${DOCUMENT_NAME}.ps
			   DEPENDS ${Document_BINARY_DIR}/${DOCUMENT_NAME}.ps
			   COMMENT "ps2pdf")

ADD_CUSTOM_TARGET(tabletReaderHelp ALL echo
	 	  DEPENDS ${Document_BINARY_DIR}/${DOCUMENT_NAME}.pdf)
INSTALL(FILES ${Document_BINARY_DIR}/${DOCUMENT_NAME}.pdf DESTINATION share/doc) 

ENDIF(PS2PDF_CONVERTER)
ENDIF(DVIPS_CONVERTER)
ENDIF(LATEX_COMPILER)
